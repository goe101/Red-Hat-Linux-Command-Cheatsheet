### Redhat Package Manager (RPM)
#### 1. Check RPM signature
	rpm --checksig <.rpm pkg>  
#### 2. Install RPM package
	rpm -ivh <.rpm pkg> 
#### 3. Check dependencies of RPM pkg	
	rpm -qpR <.rpm pkg> 
#### 4. Install RPM pkg without dependencies
	rpm -ivh --nodeps <.rpm pkg> 	
#### 5. List all files of installed RPM packages
	rpm -ql <.rpm pkg> 
#### 6. List installed RPM packages
	rpm -qa  
#### 7. List installed RPM packages
	rpm -q <.rpm pkg> 
#### 8. List all recently installed RPM packages
	rpm -qa --last  
#### 9.	Upgrade a RPM package
	rpm -Uvh <.rpm pkg> 
#### 10. Remove RPM package	
	rpm -evvnx <.rpm pkg> 
#### 11. Remove RPM package without dependencies	
	rpm -ev --nodeps 
#### 12. Query a file that belongs which RPM package	
	rpm -qf /usr/bin/htpasswd 
#### 13. Show the information of installed RPM package	
	rpm -qi vsftpd 
#### 14. Show the information of RPM package before install	
	rpm -qip <.rpm pkg> 
#### 15. Show documentation of instal RPM package	
	rpm -qdf /usr/bin/vmstat 
#### 16. Verifying a package compares information of installed files against rpm db	
	rpm -Vp <.rpm pkg> 
#### 17. Verifying all packages	
	rpm -Va
